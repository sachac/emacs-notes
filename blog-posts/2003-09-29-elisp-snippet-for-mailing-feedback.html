(defun sacha/mail-cs21a-feedback (id)
  (interactive "MID:")
  (save-excursion
    (insert "\n-------------------------------------------------------------------\n"
            "Files\n"
            "-------------------------------------------------------------------\n"
            "\n"
            "-------------------------------------------------------------------\n"
            "Vectors\n"
            "-------------------------------------------------------------------\n"
            "\n"
            "-------------------------------------------------------------------\n"
            "Other comments\n"
            "-------------------------------------------------------------------\n"))
  (insert "-------------------------------------------------------------------\n"
          "Your code\n"
          "-------------------------------------------------------------------\n")
  (insert-file-contents
   (car (file-expand-wildcards (concat "~/1/" id "/*.java") t ))))
