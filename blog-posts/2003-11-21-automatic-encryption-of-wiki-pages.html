;; crypt++ – i use this in conjunction with emacs-wiki (require ‘crypt++) (setq crypt-encryption-type ‘mcrypt

crypt-encryption-file-extension “\\(Secure\\)$\\|\\(\\.enc\\)$”) (setq emacs-wiki-ignored-extensions-regexp “\\.\\(bz2\\|gz\\|[Zz]\\|enc\\)\\'”)
